
# our algorithm 
class Reservoir:
    Y = None # the sampled path
    W = 0    # the unbiased contribution weight
    M = 0    # the effective sample count    
    F = 0    # path contribution through CV multipled by the effective sample count
    def update(self, X, w, m):
        pass

def Resample_and_CV(i, pixels: list[Pixel]) -> Reservoir:
    e = Reservoir() # create an empty reservoir    
    for j in pixels:
        # shift mapping
        Z1 = ShiftMapping(i, j.r.Y) # shift mapping from pixel j to pixel i
        Z2 = ShiftMapping(j, i.r.Y) # shift mapping from pixel i to pixel j

        # estimate the pixel difference (Eq. 16)
        diff_ij = estimate_difference(i, j, Z1, Z2)

        # adjust the path contribution via control variates (Eq. 11-13)
        e.F +=  (alpha * j.r.F + diff_ij) * j.r.W

        # evaluate the resampling weight and update reservoir e
        e.update(Z1, updated(j.r.W), j.r.M) 
    return e

# per frame, for each pixel i
def ReSTIR_CV():
    
    # 1. initialization 
    X_i, p_i, f_i = GenerateInitialSample(i) # generate an initial path sample, its pdf and its path contribution
    i.r = Reservoir(X_i, 1/p_i, 1, f_i) # store it in the reservoir

    # 2: temporal reusing
    j = CorrespondingPixelInPreviousFrame(i)
    i.r = Resample_and_CV(i,[i,j]) 
    # 3: spatial reusing
    neighbors = GetNeighboringPixels(i)
    i.r = Resample_and_CV(i,[i, *neighbors])

    # 4: final shading
    return i.r.F / i.r.M # return the adjusted path contribution
